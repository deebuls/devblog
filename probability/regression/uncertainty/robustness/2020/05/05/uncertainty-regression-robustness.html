def gauss_logL(xbar, sigma_2, mu):
  return -0.5*np.log(2*np.pi)-0.5*np.log(sigma_2)-0.5*(xbar -mu)**2/sigma_2
xbar = 1

logL -= logL.max()

x_grid, sigma_grid = np.meshgrid(diff, sigma_2)
